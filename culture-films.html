/* ================= LOAD PROPOSALS (remplacer l'ancienne fonction) ================= */ async function loadProposals(section) { const container = document.getElementById('proposals-list'); const dbg = document.getElementById('debug'); container.innerHTML = 'Chargement…'; showDebug(''); // vide l'ancien debug try { // anti-cache pour s'assurer d'avoir la dernière version const idxResp = await fetch('/content/index.json?t=' + Date.now()); if (!idxResp.ok) throw new Error('index.json introuvable (status ' + idxResp.status + ')'); const idxText = await idxResp.text(); let idx; try { idx = JSON.parse(idxText); } catch (e) { // affiche le texte brute si ce n'est pas du JSON valide showDebug('index.json is not valid JSON — raw content:\n\n' + idxText); throw new Error('index.json invalide'); } // debug: montre un extrait de index.json showDebug('index.json chargé — ' + (Array.isArray(idx) ? idx.length : 'non-array') + ' entrées\n\n' + JSON.stringify(idx.slice(0, 10), null, 2)); // filtre et trie (les plus récents en tête si created_at présent) let items = (Array.isArray(idx) ? idx : []).filter(i => String(i.section) === String(section)); items.sort((a, b) => { const ta = a.created_at ? Date.parse(a.created_at) : 0; const tb = b.created_at ? Date.parse(b.created_at) : 0; return tb - ta; }); if (!items.length) { container.innerHTML = '

Aucune proposition.

'; return; } // charge chaque proposition, tolère les erreurs par item const htmlItems = await Promise.all(items.map(async it => { try { // calcule l'URL du fichier markdown let fileUrl = it.path || it.url || ''; if (!fileUrl) throw new Error('path manquant dans index.json pour cette entrée'); // si path n'est pas une URL absolue, rendre relatif au site if (!/^https?:\/\//i.test(fileUrl)) { fileUrl = fileUrl.startsWith('/') ? fileUrl : '/' + fileUrl; } // anti-cache pour le fetch du md const mdResp = await fetch(fileUrl + '?t=' + Date.now()); if (!mdResp.ok) { throw new Error(`Fichier introuvable (${mdResp.status}) -> ${fileUrl}`); } let md = await mdResp.text(); // supprime frontmatter YAML si présent (robuste) md = md.replace(/^---\s*[\s\S]*?\s*---\s*/m, ''); // parse markdown en HTML (avec gestion d'erreur) let contentHtml = ''; try { contentHtml = marked.parse(md); } catch (e) { console.error('Erreur parse markdown:', e); contentHtml = '
Erreur de parsing markdown
'; } // titre prioritaire : it.title, sinon titre extrait du markdown, sinon basename du path const title = it.title || (md.match(/^#\s*(.+)$/m) ? md.match(/^#\s*(.+)$/m)[1].trim() : (fileUrl.split('/').pop() || 'Sans titre')); // format date lisible const created = it.created_at ? new Date(it.created_at).toLocaleString() : ''; return `

${escapeHtml(title)}

Par ${escapeHtml(it.pseudo || 'Anonyme')}${created ? ' — ' + escapeHtml(created) : ''}
${contentHtml}
`; } catch (itemErr) { console.warn('Erreur item:', itemErr); // on retourne un message visible pour cet item (plutôt que de casser toute la page) return `

Proposition invalide

Erreur: ${escapeHtml(String(itemErr.message || itemErr))}
`; } })); container.innerHTML = htmlItems.join('\n'); } catch (err) { container.innerHTML = '

Erreur de chargement.

'; // affiche l'erreur dans le debug pour que tu puisses copier/coller showDebug((document.getElementById('debug').textContent || '') + '\n\nErreur globale: ' + String(err)); console.error('loadProposals error', err); } }
